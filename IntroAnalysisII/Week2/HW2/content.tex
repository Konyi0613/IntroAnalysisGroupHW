\begin{problem}
    Let $GL(n, \mathbf{R}) \subset \mathbf{R}^{n \times n}$ denote the general linear group of invertible $n \times n$ real matrices. This set is an open subset of the vector space $\mathbf{R}^{n \times n}$. Consider the inversion map $g: GL(n, \mathbf{R}) \to GL(n, \mathbf{R})$ defined by$$g(A) = A^{-1}.$$Prove that $g$ is  differentiable at any $A \in GL(n, \mathbf{R})$, and show that its total derivative $g'(A): \mathbf{R}^{n \times n} \to \mathbf{R}^{n \times n}$ is the linear map given by$$g'(A)(H) = -A^{-1} H A^{-1}.$$\\
 Hints:Algebraic Identity: For a small perturbation matrix $H$, start by factoring $A$ out of the expression $(A+H)^{-1}$ to relate it to the identity matrix:$$(A+H)^{-1} = (A(I + A^{-1}H))^{-1} = (I + A^{-1}H)^{-1} A^{-1}.$$ The Geometric Series: Recall that for a matrix $M$ with $\|M\| < 1$, the following power series (the Neumann series) converges:$$(I + M)^{-1} = I - M + M^2 - M^3 + \dots$$Apply this to $M = A^{-1}H$ for $H$ sufficiently small. Isolate the Linear Term: Expand $g(A+H) - g(A)$ using the series. Identify the term that is linear in $H$. This is your candidate for $g'(A)(H)$.
\end{problem}

\begin{proof}
    To find the derivative of $g$ at $A$, we apply a small matrix perturbation $H \in \mathbb{R}^{n \times n}$ and expand $g(A+H)$:
    \[
        g(A+H) = (A+H)^{-1} = (A(I+A^{-1}H))^{-1} = (I+A^{-1}H)^{-1}A^{-1}
    \]
    And notice that for a small matrix with $\| M \| < 1$, the following power series converges:
    \[
        (I + M)^{-1} = I - M + M^2 - M^3 + \dots,
    \]
    Apply this to $M = A^{-1}H$ since $H$ is small enough matrix perturbation and $\| A \| < \infty$, we can get:
    \[
        g(A+H) = (I+A^{-1}H)^{-1}A^{-1} = (I-A^{-1}H+(A^{-1}H)^2-\dots)A^{-1}.
    \]
    Hence, we have
    \[
        \begin{aligned}
            g(A+H)-g(A) &= (-A^{-1}H+(A^{-1}H)^2-\dots)A^{-1} \\ 
            &= -A^{-1}HA^{-1} + ((A^{-1}H)^2-(A^{-1}H)^3 +\dots)A^{-1}.
        \end{aligned}
    \]
    Hence, we may choose $L: \mathbb{R}^{n \times n} \to \mathbb{R}^{n \times n}$ by
    \[
        L(H) = -A^{-1}HA^{-1}
    \]
    Thus, we have
    \[
        \begin{aligned}
            \lim_{H \to 0} \frac{\|g(A+H) - g(A) - L(H) \|}{\| H \|} &= \lim_{H \to 0} \frac{\| ((A^{-1}H)^2-(A^{-1}H)^3 +\dots )A^{-1}\|}{\| H\|} \\
            &\le \lim_{H \to 0} \frac{(\sum_{k=2}^{\infty}\| A^{-1} \|^k\| H\|^k)\|A^{-1}\|}{\| H\|} \\
            &= \lim_{H \to 0} (\sum_{k=1}^{\infty}\| A^{-1} \|^{k+1}\| H\|^k)\|A^{-1}\| \\
            &= \lim_{H \to 0} (\sum_{k=1}^{\infty}(\| A^{-1} \|\| H\|)^k)\|A^{-1}\|^2
        \end{aligned}
    \]
    And since $\|A\| < \infty$ and $\| H\| \to 0$, $\lvert \| A\|\|H\| \rvert< 1$, so we can use the geometric identity:
    \[
        \begin{aligned}
            0 \le \lim_{H \to 0} \frac{\|g(A+H) - g(A) - L(H) \|}{\| H \|} &\le \lim_{H \to 0} (\sum_{k=1}^{\infty}(\| A^{-1} \|\| H\|)^k)\|A^{-1}\|^2 \\
            &= \lim_{H \to 0}\frac{\| A^{-1}\|\|H\|}{1-\| A^{-1}\|\|H\|} \cdot \| A^{-1}\|^2 \\
            &= 0 \cdot \| A^{-1}\|^2 = 0,
        \end{aligned}
    \]
    and hence by squeeze theorem,
    \[
        \lim_{H \to 0} \frac{\|g(A+H) - g(A) - L(H) \|}{\| H \|} = 0.
    \]
    So $g$ is differentiable at any $A$, and the total derivative $g'(A)$ is the linear map given by:
    \[
        g'(A)(H) = L(H) = -A^{-1}HA^{-1}
    \]
\end{proof}

\begin{problem}
    Let $f : \mathbf{R}^2 \to \mathbf{R}$ be defined by
$$
f(x,y)
:=
\frac{xy^3}{x^2+y^2}
\quad \text{when } (x,y)\ne(0,0),
\qquad
f(0,0):=0.
$$
\begin{itemize}
\item[(a)] Find $f_x(0,0)$ and $f_y(0,0)$.
 \item[(b)] Prove that $f$ is differentiable everywhere and that $\nabla f$ is continuous on $\mathbf{R}^2$, i.e.\ $f\in C^1(\mathbf{R}^2)$. Hint: You can use polar coordinate 
$x=r \cos(\theta), y=r \sin (\theta)$ when you try to show that  $f_x \to 0$ and $f_y \to 0$ as
$(x,y)\to 0$. Also $(x,y)\to 0$ implies $r\to 0$
\item[(c)] Show that the mixed partial derivatives $f_{xy}(0,0)$ and $f_{yx}(0,0)$ both exist.
\item[(d)] Compute $f_{xy}(0,0)$ and $f_{yx}(0,0)$ and show that they are \emph{not} equal.
\item[(e)] Identify precisely which hypothesis of Clairaut's theorem fails here.
\end{itemize}
\end{problem}

\begin{proof}[(a)]
    By definition,
    \[
    f_x(0,0) = \lim_{h \to 0} \frac{f(h,0) - f(0,0)}{h} = \lim_{h \to 0} \frac{\frac{h \cdot0^3}{h^2+0^2}-0}{h} = \lim_{h \to 0} \frac{0 - 0}{h} = 0.
    \]
    Simlilarly,
    \[
    f_y(0,0) = \lim_{k \to 0} \frac{f(0,k) - f(0,0)}{k} = \lim_{k \to 0} \frac{\frac{0\cdot k^3}{0^2+k^2}-0}{k} = \lim_{k \to 0} \frac{0 - 0}{k} = 0.
    \]
    Hence,
    \[
    f_x(0,0)=f_y(0,0)=0.
    \]
\end{proof}

\begin{proof}[(b)]
    By (a), both $f_x(0,0)$ and $f_y(0,0)$ exist and equal 0. For $(x,y)\neq(0,0)$, we have
    \[
    f_x(x,y) = \frac{y^3(x^2+y^2) - xy^3(2x)}{(x^2+y^2)^2} = \frac{y^3(y^2-x^2)}{(x^2+y^2)^2}
    \]
    and
    \[
    f_y(x,y) = \frac{3xy^2(x^2+y^2) - xy^3(2y)}{(x^2+y^2)^2} = \frac{xy^2(3x^2+y^2)}{(x^2+y^2)^2}.
    \]
    To show the continuity at $(0,0)$, we use polar coordinates:
    \[
    f_x(r,\theta) = \frac{r^5 \sin^3\theta (\sin^2\theta - \cos^2\theta)}{r^4} = r \sin^3\theta (\sin^2\theta - \cos^2\theta).
    \]
    As $(x,y) \to (0,0)$, $r \to 0$. Since $|\sin^3\theta (\sin^2\theta - \cos^2\theta)| \le 2$, we have $|f_x| \le 2r$. By the Squeeze Theorem, $\lim_{r \to 0} f_x(r,\theta) = 0 = f_x(0,0)$. Similarly for $f_y$, we have
    \[
    f_y(r,\theta) = \frac{r^5 \cos\theta \sin^2\theta (3\cos^2\theta + \sin^2\theta)}{r^4} = r \cos\theta \sin^2\theta (3\cos^2\theta + \sin^2\theta).
    \]
    Since $|\cos\theta \sin^2\theta (3\cos^2\theta + \sin^2\theta)| \le 4$, we have $|f_y| \le 4r$, so $\lim_{r \to 0} f_y(r,\theta) = 0 = f_y(0,0)$. Because all partial derivatives exist and are continuous everywhere on $\mathbb{R}^2$, by Theorem 6.3.8, we can conclude that $f$ is differentiable everywhere.
\end{proof}

\begin{proof}[(c)]
    By definition, 
    \[
    f_{xy}(0,0) = \lim_{y \to 0} \frac{f_x(0,y) - f_x(0,0)}{y} = \lim_{y \to 0} \frac{\frac{y^3(y^2-0)}{(0+y^2)^2} - 0}{y} = \lim_{y \to 0} \frac{y}{y} = 1
    \]
    and
    \[
    f_{yx}(0,0) = \lim_{x \to 0} \frac{f_y(x,0) - f_y(0,0)}{x} = \lim_{x \to 0} \frac{\frac{x(0)(3x^2+0)}{(x^2+0)^2} - 0}{x} = \lim_{x \to 0} \frac{0}{x} = 0.
    \]
    Hence, both exist.
\end{proof}

\begin{proof}[(d)]
    By (c), we compute that $f_{xy}(0,0)=1$ and $f_{yx}(0,0)=0$, so they are not equal.
\end{proof}

\begin{proof}[(e)]
    For $(x,y) \neq (0,0)$, we have:
    \[
    f_{xy}(x,y) = \frac{\partial}{\partial y} \frac{y^3(y^2-x^2)}{(x^2+y^2)^2} = \frac{y^2(y^4+6x^2y^2-3x^4)}{(x^2+y^2)^3}.
    \]
    Now, we examine the limit of $f_{xy}(x,y)$ as $(x,y) \to (0,0)$ using different ways. We have:
    \[
    \lim_{y \to 0} f_{xy}(0,y) = \lim_{y \to 0} \frac{y^6}{y^6} = 1, 
    \]
    and we also have:
    \[
    \lim_{x \to 0} f_{xy}(x,x) = \lim_{x \to 0} \frac{4x^6}{8x^6} = \frac{1}{2}.
    \]
    We can conclude that $\lim_{(x,y) \to (0,0)}f_{xy}(x, y)$ does not exist. Hence, $f_{xy}$ is not continuous at $(0,0)$, implying that $f$ is not twice continuously differentiable, and thus the hypothesis fails.
\end{proof}

\begin{problem}

Fix $(x_0,y_0)\in E$.  For sufficiently small $h,k$ such that the rectangle
\[
R_{h,k}:=[x_0,x_0+h]\times [y_0,y_0+k]\subset E,

define the second finite difference

\Delta_{h,k} f

and conclude $f_{xy}(x_0,y_0)=f_{yx}(x_0,y_0)$. Hint:  If $F$ is continuous on $[x_0,x_0+h]\times [y_0,y_0+k]$, then there exists

\end{problem}

\begin{proof}[(a)]
    By applying Fundamental Theorem of Calculus, we have 
    \begin{align*}
        \int _{x_0}^{x_0 + h} \int _{y_0}^{y_0 + k} f_{xy}(x, y) \mathrm{d} y \mathrm{d} x &= \int _{x_0}^{x_0 + h} \left. f_x(x, y) \right]_{y_0}^{y_0 + k} \mathrm{d} x = \int _{x_0}^{x_0 + h} f_x (x, y_0 + k) - f_x(x, y_0) \mathrm{d} x \\
        &= \left[ f(x, y_0 + k) - f(x, y_0) \right]_{x_0}^{x_0 + h} \\
        &= \left( f(x_0 + h, y_0 + k) - f(x_0 + h, y_0) \right) - \left( f(x_0, y_0 + k) - f(x_0, y_0) \right) \\
        &= f(x_0 + h, y_0 + k) - f(x_0 + h, y_0) - f(x_0, y_0 + k) + f(x_0, y_0) \\
        &= \Delta _{h, k} f      
    \end{align*}    
    and 
    \begin{align*}
        \int _{y_0}^{y_0 + k} \int _{x_0}^{x_0 + h} f_{yx}(x, y) \mathrm{d} x \mathrm{d} y &= \int _{y_0}^{y_0 + k} \left. f_y(x, y) \right]_{x_0}^{x_0 + h} \mathrm{d} y = \int _{y_0}^{y_0 + k} f_y (x_0 + h, y) - f_y(x_0, y) \mathrm{d} y \\
        &= \left[ f(x_0 + h, y) - f(x_0, y) \right]_{y_0}^{y_0 + k} \\
        &= f(x_0 + h, y_0 + k) - f(x_0 + h, y_0) - f(x_0, y_0 + k) + f(x_0, y_0) \\
        &= \Delta _{h, k} f.      
    \end{align*}
    Thus,
    \[
        \Delta_{h,k} f
=
\int_{x_0}^{x_0+h}\int_{y_0}^{y_0+k} f_{xy}(x,y)\,dy\,dx
=
\int_{y_0}^{y_0+k}\int_{x_0}^{x_0+h} f_{yx}(x,y)\,dx\,dy.
    \]
\end{proof}

\begin{proof}[(b)]
    We know \(R_{h, k}\) is closed and bounded on \(\mathbb{R} ^2\), so \(R_{h, k}\) is compact by Heine Borel's theorem. Also, for sufficiently small \(h, k\), we know \(R_{h, k} \subseteq E\) since \(E\) is open, so in this case \(f_{xy}\) is continuous on \(R_{h, k}\). Now since \(f_{xy}\) is continuous on \(R_{h, k}\) and \(R_{h, k}\) is compact, so \(\exists v_{\text{max}}, v_{\text{min}} \in R_{h, k}\) s.t. 
    \[
        f_{xy}(v_{\text{max}}) = \max _{v \in R_{h, k}} f_{xy}(v), \quad f_{xy} (v_{\text{min}}) = \min_{v \in R_{h, k}} f_{xy}(v). 
    \]          
    Now let \(f_{xy}(v_{\text{max}}) = M\) and \(f_{xy} (v_{\text{min}}) = m\), then for all \(v \in R_{h, k}\), \(m \le f_{xy}(v) \le M\). Thus, we have 
    \[
        hk \cdot m \le \int _{x_0}^{x_0 + h} \int _{y_0}^{y_0 + k} f_{xy}(x, y) \mathrm{d} y \mathrm{d} x \le hk \cdot M.  
    \]    
    Note that \(R_{h, k}\) is connected and \(f_{xy}\) is continuous, so \(f_{xy} (R_{h, k})\) is a connected subset of \(\mathbb{R} \), i.e. \(f_{xy} (R_{h, k}) = [m, M]\), so there exists \(\left( \overline{x} , \overline{y}  \right) \in R_{h, k} \) s.t. 
    \[
        f_{xy} \left( \overline{x} , \overline{y}  \right) = \frac{1}{hk} \int _{x_0}^{x_0 + h} \int _{y_0}^{y_0 + k} f_{xy} (x, y) \mathrm{d} y \mathrm{d} x = \frac{1}{hk} \Delta _{h, k} f.   
    \]      
    Now if \((h, k) \to (0, 0)\), then \((\overline{x} , \overline{y} ) \to (x_0, y_0)\), and since \(f_{xy}\) is continuous on \(R_{h, k}\), we have \(f_{xy} \left( \overline{x} , \overline{y}  \right) \to f_{xy}(x_0, y_0) \). Thus, we can conclude 
    \[
        \frac{1}{hk} \Delta _{h, k} f \to f_{xy} (x_0, y_0) \text{ as } (h, k) \to (0, 0). 
    \]
    By using same arguments but replacing \(f_{xy}\) by \(f_{yx}\) and use 
    \[
        \Delta_{h,k} f = \int_{y_0}^{y_0+k}\int_{x_0}^{x_0+h} f_{yx}(x,y)\,dx\,dy,
    \]       
    we can deduce 
    \[
        \frac{1}{hk} \Delta _{h, k} f \to f_{yx} (x_0, y_0) \text{ as } (h, k) \to (0, 0).
    \]
    Finally, we can conclude 
    \[
        f_{xy}(x_0, y_0) = f_{yx} (x_0, y_0).
    \]
\end{proof}

\begin{problem}
    Let $f$ and $g$ be functions from $\mathbf{R}^n$ to $\mathbf{R}^m$. Assume that $f$ is
differentiable at $c$, that $f(c)=0$, and that $g$ is continuous at $c$.
Let
\[
h(x)=g(x)\cdot f(x).
\]
Prove that $h$ is differentiable at $c$ and that
\[
h'(c)(u)=g(c)\cdot\bigl(f'(c)(u)\bigr)
\qquad \text{if } u\in \mathbf{R}^n.
\] Note that $g(x)\cdot f(x)$ is the inner product of two vectors in $\mathbf{R}^m$ and $h$ is a function from
$\mathbf{R}^n$ to $\mathbf{R}$.
\end{problem}

\begin{proof}
    We want to show that there exist a linear transformation $L(h)$
    such that
    \[
        \lim_{t \to 0} \frac{\| h(c+t) - h(c) -L(t)\|}{\| t\|} = 0
    \]
    We pick 
    \[
        L(t) = g(c) \cdot (f'(c)(t))
    \]
    and show the limit is indeed equal to $0$.

    \[
        \begin{aligned}
            \lim_{t \to 0} \frac{\| h(c+t) - h(c) -L(t)\|}{\| t\|} &= \lim_{t \to 0} \frac{\| g(c+t) \cdot f(c+t) - g(c) \cdot f(c) -L(t)\|}{\| t\|} \\
            &= \lim_{t \to 0} \frac{\| g(c+t) \cdot f(c+t)  -L(t)\|}{\| t\|} ,\qquad\text{because $f(c) = 0.$}
        \end{aligned}
    \]
    
    Notice that $f$ is differentiable at $0$, so
    \[
        \lim_{t \to 0} \frac{\| f(c+t) - f(c) - f'(c)(t)\|}{\| t\|} = 0,
    \]
    and since $f(c) = 0$, so
    \[
        \lim_{t \to 0} \frac{\|f(c+t) - f'(c)(t)\|}{\|t\|} = 0.
    \]
    Then we plug the result back into the equation, we can get
    \[
        \begin{aligned}
            \lim_{t \to 0} \frac{\| h(c+t) - h(c) -L(t)\|}{\| t\|} 
            &= \lim_{t \to 0} \frac{\| g(c+t) \cdot f(c+t)  -L(t)\|}{\| t\|} \\
            &= \lim_{t \to 0} \frac{\| g(c+t) \cdot f(c+t)  - g(c) \cdot f(c+t) + g(c)\cdot f(c+t)  -g(c) \cdot (f'(c)(t))\|}{\| t\|} \\
            &= \lim_{t \to 0} \frac{\| (g(c+t)-g(c)) \cdot f(c+t) +g(c) \cdot (f(c+t)-f'(c)(t))\|}{\| t\|} \\
            &\le \lim_{t \to 0} \frac{\| (g(c+t)-g(c)) \cdot f(c+t) \|}{\| t\|} + \lim_{t \to 0} \frac{\|g(c) \cdot (f(c+t)-f'(c)(t))\|}{\| t\|}\\
            &\le \lim_{t \to 0} \frac{\| (g(c+t)-g(c)) \| \|f(c+t) \|}{\| t\|} + \lim_{t \to 0} \frac{\|g(c)\|\|(f(c+t)-f'(c)(t))\|}{\| t\|} \\
            &= \lim_{t \to 0} \| (g(c+t)-g(c))  \| \cdot \frac{ \|f(c+t) \|}{\| t\|} + \|g(c)\| \cdot \lim_{t \to 0} \frac{\|(f(c+t)-f'(c)(t))\|}{\| t\|} \\
            &= \lim_{t \to 0} \| (g(c+t)-g(c))  \| \cdot \frac{ \|f(c+t) \|}{\| t\|} + \|g(c)\| \cdot  0 \quad\text{from previous part} \\
            &= \lim_{t \to 0} \| (g(c+t)-g(c))  \| \cdot \frac{ \|f(c+t) - f'(c)(t) + f'(c)(t)\|}{\| t\|} \\
            &\le \lim_{t \to 0} \| (g(c+t)-g(c))  \| \cdot \frac{ \|f(c+t) - f'(c)(t) \| + \| f'(c)(t)\|}{\| t\|} 
        \end{aligned}
    \]
    Then notice that $g$ is continuous at $c$, by definition, we know that
    \[
         \forall \varepsilon >0, \exists \delta > 0 \text{ such that }\|x-x_0\| < \delta \implies \|g(x) - g(x_0)\| < \varepsilon.
    \]
    and since 
    \[
        \lim_{t \to 0} \| (c+t) - c \| \to 0,
    \]
    so 
    \[
        \lim_{t \to 0} \| (g(c+t)-g(c)) \| \to 0.
    \]
    and since 
    \[
        \lim_{t \to 0} \frac{ \|f(c+t) - f'(c)(t) \| }{\| t\|} = 0
    \]
    and
    \[
        \lim_{t \to 0} \frac{ \|f'(c)(t) \| }{\| t\|} \le \lim_{t \to 0} \frac{ \|t\|\|f'(c)(1) \| }{\| t\|} = \|f'(c)(1) \| <\infty
    \]
    so 
    \[
        \frac{ \|f(c+t) - f'(c)(t) \| + \| f'(c)(t)\|}{\| t\|} \text{ is bounded.}
    \]
    And hence
    \[
        \begin{aligned}
            \lim_{t \to 0} \frac{\| h(c+t) - h(c) -L(t)\|}{\| t\|} 
            &\le \lim_{t \to 0} \| (g(c+t)-g(c))  \| \cdot \frac{ \|f(c+t) - f'(c)(t) \| + \| f'(c)(t)\|}{\| t\|} \\
            &\le 0 \cdot \text{bounded term} = 0.
        \end{aligned}
    \]
    So by definition, $h$ is differentiable at $c$ and that 
    \[
        h'(c)(u) = L(u) = g(c)\cdot(f'(c)(u))
    \]
\end{proof}

\begin{problem}
    Let $f$ and $g$ be real-valued functions defined on $\mathbf{R}$ with continuous second
derivatives $f''$ and $g''$. Define
\[
F(x,y)=f\bigl(x+g(y)\bigr)
\qquad \text{for each }(x,y)\in \mathbf{R}^2.
\]
Find formulas for all first- and second-order partial derivatives of $F$ in terms of the
derivatives of $f$ and $g$. Verify the relation
\[
F_x\,F_{xy}=F_y\,F_{xx}.
\]
\end{problem}

\begin{proof}
    For first-order partial derivatives, we have:
    \begin{itemize}
        \item $F_x(x,y) = f'(x+g(y)) \cdot \frac{\partial}{\partial x}(x+g(y)) = f'(x+g(y))$
        \item $F_y(x,y) = f'(x+g(y)) \cdot \frac{\partial}{\partial y}(x+g(y)) = f'(x+g(y))g'(y)$
    \end{itemize}
    For second-order partial derivatives, we have:
    \begin{itemize}
        \item $F_{xx}(x,y) = \frac{\partial}{\partial x}[f'(x+g(y))] = f''(x+g(y))$
        \item $F_{yy}(x,y) = \frac{\partial}{\partial y}[f'(x+g(y))g'(y)] = f''(x+g(y))[g'(y)]^2 + f'(x+g(y))g''(y)$
        \item $F_{xy}(x,y) = \frac{\partial}{\partial y}[f'(x+g(y))] = f''(x+g(y))g'(y)$
        \item $F_{yx}(x,y) = \frac{\partial}{\partial x}[f'(x+g(y))g'(y)] = f''(x+g(y))g'(y)$
    \end{itemize}
    Thus, we have
    \[
    F_x F_{xy} = [f'(x+g(y))] \cdot [f''(x+g(y))g'(y)]
    \]
    and
    \[
    F_y F_{xx} = [f'(x+g(y))g'(y)] \cdot [f''(x+g(y))].
    \]
    By comparing the two formulas, we get
    \[
    F_x F_{xy} = F_y F_{xx}.
    \]
\end{proof}
