\[

\]
define the second finite difference
\[
\Delta_{h,k} f
:=
f(x_0+h,y_0+k)-f(x_0+h,y_0)-f(x_0,y_0+k)+f(x_0,y_0).
\]
\begin{itemize}
\item[(a)] Prove the identity
\[
\Delta_{h,k} f
=
\int_{x_0}^{x_0+h}\int_{y_0}^{y_0+k} f_{xy}(x,y)\,dy\,dx
=
\int_{y_0}^{y_0+k}\int_{x_0}^{x_0+h} f_{yx}(x,y)\,dx\,dy.
\]
